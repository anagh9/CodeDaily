The 3n + 1 Problem : Consider the following algorithm to generate a sequence of numbers. Start with an integer n. If n is even, divide by 2. If n is odd, multiply by 3 and add 1. Repeat this process with the new value of n, terminating when n = 1. For example, the following sequence of numbers will be generated for n = 22:
22 11 34 17 52 26 13 40 20 10 5 16 8 4 2 1

So Given Two Input i, j and We have to find output maximum cyclesize of getting output 

        
  int cycleSize(int x) {
	int cycle = 1;
	while (x != 1) {
		if (x % 2 == 0) {
			x = x / 2;
		} else {
			x = x * 3 + 1;
		}
		++cycle;
	}
	return cycle;
}

int maxCycleSize(int a, int b) {
	if (a > b) swap(a, b);

	int maxCycle = 0;
	for (; a <= b; a++) {
		int thisCycleSize = cycleSize(a);
		maxCycle = max(maxCycle, thisCycleSize);
	}
	return maxCycle;
}
