 int cycleSize(int x) {
	int cycle = 1;
	while (x != 1) {
		if (x % 2 == 0) {
			x = x / 2;
		} else {
			x = x * 3 + 1;
		}
		++cycle;
	}
	return cycle;
}

int maxCycleSize(int a, int b) {
	if (a > b) swap(a, b);

	int maxCycle = 0;
	for (; a <= b; a++) {
		int thisCycleSize = cycleSize(a);
		maxCycle = max(maxCycle, thisCycleSize);
	}
	return maxCycle;
}
